years = salaries.groupby(['fiscalYear'])['grossPay'].sum().reset_index().sort_values('grossPay', ascending=False)[:10]

## Plotting 
f, ax = plt.subplots(figsize=(12, 5))
ax = sns.barplot(data=years, x = 'fiscalYear', y = 'grossPay', color="#1d3557")
plt.xlabel("Total Gross Pay")
plt.ylabel("Fiscal Year")
plt.show()

## Calculating the total salaries by each agency
agencies = salaries.groupby(['agencyName'])['grossPay'].sum().reset_index().sort_values('grossPay', ascending=False)

## changing the scientific format
agencies['grossPay'] = agencies.grossPay.astype('int64')

## Plotting 
f, ax = plt.subplots(figsize=(14, 15))
ax = sns.barplot(data=agencies, y = 'agencyName', x = 'grossPay', palette="Blues_d")
plt.show()
